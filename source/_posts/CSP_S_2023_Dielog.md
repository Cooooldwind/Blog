附 T2 代码（非考时（FJ怎么还没有\~\~\~））

```cpp
#include<bits/stdc++.h>
using namespace std;

int n,cnt;
string a;

bool isclear(string a){
    while(a.length()){
        bool flag = true;
        for(int i = 0;i < a.length();i++){
            if(a[i] == a[i + 1]){
                string nxt = "";
                flag = false;
                for(int j = 0;j <= a.length();j++){
                    if(j != i && j != i + 1) nxt = nxt + a[j];
                }
                a = nxt;
            }
        }
        if(flag) return false;
    }
    return a.length() == 0;
}

int main(){
    cin >> n >> a;
    int cnt = 0;
    for(int i = 2;i <= n;i += 2){
        for(int j = 0;j < a.length();j++){
            if(j + i > n) break;
            if(isclear(a.substr(j,i))) cnt++;
        }
    }
    cout << cnt;
    return 0;
}
```
